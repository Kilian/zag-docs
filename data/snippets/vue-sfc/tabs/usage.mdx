```jsx
<script setup>
import * as tabs from "@zag-js/tabs";
import { normalizeProps, useMachine, useSetup } from "@zag-js/vue";
import { computed } from "vue";

const data = [
  { value: "item-1", label: "Item one", content: "Item one content" },
  { value: "item-2", label: "Item two", content: "Item two content" },
  { value: "item-3", label: "Item three", content: "Item three content" },
];

const [state, send] = useMachine(tabs.machine({ value: "item-1" }));
const ref = useSetup({ send, id: "1" });
const api = computed(() => tabs.connect(state.value, send, normalizeProps));
</script>

<template>
  <div ref="ref" v-bind="api.rootProps">
    <div v-bind="api.tablistProps">
      <button
        v-for="item in data"
        v-bind="api.getTabProps({ value: item.value })"
        :key="item.value"
      >
        {item.label}
      </button>
    </div>
    <div
      v-for="item in data"
      v-bind="api.getTabPanelProps({ value: item.value })"
      key="{item.value}"
    >
      <p>{item.content}</p>
    </div>
  </div>
</template>
```
